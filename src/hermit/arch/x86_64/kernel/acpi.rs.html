16/// Memory at this physical address is supposed to contain a pointer to the Extended BIOS Data Area (EBDA).
17const EBDA_PTR_LOCATION: PhysAddr = PhysAddr::new(0x0000_040e);
18/// Minimum physical address where a valid EBDA must be located.
19const EBDA_MINIMUM_ADDRESS: PhysAddr = PhysAddr::new(0x400);
20/// The size of the EBDA window that is searched for an ACPI RSDP.
21const EBDA_WINDOW_SIZE: usize = 1024;
22/// The lower bound of the other address range, where the ACPI RSDP could be located.
23const RSDP_SEARCH_ADDRESS_LOW: PhysAddr = PhysAddr::new(0xe_0000);
24/// The upper bound of the other address range, where the ACPI RSDP could be located.
25const RSDP_SEARCH_ADDRESS_HIGH: PhysAddr = PhysAddr::new(0xf_ffff);
26/// Length in bytes of the structure, over which the basic (ACPI 1.0) checksum is calculated.
27const RSDP_CHECKSUM_LENGTH: usize = 20;
28/// Length in byte sof the structure, over which the extended (ACPI 2.0+) checksum is calculated.
29const RSDP_XCHECKSUM_LENGTH: usize = 36;

93/// A convenience structure to work with an ACPI table.
94/// Maps a single table to memory and frees the memory when a variable of this structure goes out of scope.
95#[derive(Debug)]
96pub struct AcpiTable<'a> {
97	header: &'a AcpiSdtHeader,
98	allocated_virtual_address: VirtAddr,
99	allocated_length: usize,
100}
101
102impl AcpiTable<'_> {
103	fn map(physical_address: PhysAddr) -> Self {
104		if env::is_uefi() {
105			// For UEFI Systems, the tables are already mapped so we only need to return a proper reference to the table
106			let allocated_virtual_address = VirtAddr::new(physical_address.as_u64());
107			let header = unsafe {
108				allocated_virtual_address
109					.as_ptr::<AcpiSdtHeader>()
110					.as_ref()
111					.unwrap()
112			};
113			let allocated_length = usize::try_from(header.length).unwrap();
114
115			return Self {
116				header,
117				allocated_virtual_address,
118				allocated_length,
119			};
120		}
121
122		let mut flags = PageTableEntryFlags::empty();
123		flags.normal().read_only().execute_disable();
124
125		// Allocate two 4 KiB pages for the table and map it.
126		// This guarantees that we can access at least the "length" field of the table header when its physical address
127		// crosses a page boundary.
128		let mut allocated_length = 2 * BasePageSize::SIZE as usize;
129		let mut count = allocated_length / BasePageSize::SIZE as usize;
130
131		let physical_map_address = physical_address.align_down(BasePageSize::SIZE);
132		let offset = (physical_address - physical_map_address) as usize;
133		let mut virtual_address = virtualmem::allocate(allocated_length).unwrap();
134		paging::map::<BasePageSize>(virtual_address, physical_map_address, count, flags);
135
136		// Get a pointer to the header and query the table length.
137		let mut header_ptr: *const AcpiSdtHeader = (virtual_address + offset).as_ptr();
138		let table_length = unsafe { (*header_ptr).length } as usize;
139
140		// Remap if the length exceeds what we've allocated.
141		if table_length > allocated_length - offset {
142			virtualmem::deallocate(virtual_address, allocated_length);
143
144			allocated_length = (table_length + offset).align_up(BasePageSize::SIZE as usize);
145			count = allocated_length / BasePageSize::SIZE as usize;
146
147			virtual_address = virtualmem::allocate(allocated_length).unwrap();
148			paging::map::<BasePageSize>(virtual_address, physical_map_address, count, flags);
149
150			header_ptr = (virtual_address + offset).as_ptr();
151		}
152
153		// Return the table.
154		Self {
155			header: unsafe { &*header_ptr },
156			allocated_virtual_address: virtual_address,
157			allocated_length,
158		}
159	}
160
161	pub fn header_start_address(&self) -> usize {
162		ptr::from_ref(self.header).addr()
163	}
164
165	pub fn table_start_address(&self) -> usize {
166		self.header_start_address() + mem::size_of::<AcpiSdtHeader>()
167	}
168
169	pub fn table_end_address(&self) -> usize {
170		self.header_start_address() + self.header.length as usize
171	}
172}
173
174impl Drop for AcpiTable<'_> {
175	fn drop(&mut self) {
176		if !env::is_uefi() {
177			virtualmem::deallocate(self.allocated_virtual_address, self.allocated_length);
178		}
179	}
180}

256/// Verifies the checksum of an ACPI table.
257/// Tables supporting this feature contain a "checksum" field. The value of this field is chosen, so that a
258/// (wrapping) sum over all table fields equals zero.
259fn verify_checksum(start_address: usize, length: usize) -> Result<(), ()> {
260	// Get a slice over all bytes of the structure that are considered for the checksum.
261	let slice =
262		unsafe { slice::from_raw_parts(ptr::with_exposed_provenance(start_address), length) };
263
264	// Perform a wrapping sum over these bytes.
265	let checksum = slice.iter().fold(0, |acc: u8, x| acc.wrapping_add(*x));
266
267	// This sum must equal to zero to be valid.
268	if checksum == 0 { Ok(()) } else { Err(()) }
269}

271/// Tries to find the ACPI RSDP within the specified address range.
272/// Returns a reference to it within the Ok() if successful or an empty Err() on failure.
273fn detect_rsdp(start_address: PhysAddr, end_address: PhysAddr) -> Result<&'static AcpiRsdp, ()> {
274	// Trigger page mapping in the first iteration!
275	let mut current_page = 0;
276
277	// Look for the ACPI RSDP in all possible 16-byte aligned addresses within this range.
278	for current_address in (start_address.as_usize()..end_address.as_usize()).step_by(16) {
279		// Have we crossed a page boundary in the last iteration?
280		if current_address / BasePageSize::SIZE as usize > current_page {
281			// Identity-map this possible page of the RSDP.
282			let frame = PhysFrame::<BasePageSize>::containing_address(x86_64::PhysAddr::new(
283				current_address as u64,
284			));
285			paging::identity_map::<BasePageSize>(frame.start_address().into());
286			current_page = current_address / BasePageSize::SIZE as usize;
287		}
288
289		// Verify the signature to find out if this is really an ACPI RSDP.
290		let rsdp = unsafe { &*(ptr::with_exposed_provenance::<AcpiRsdp>(current_address)) };
291		if &rsdp.signature != b"RSD PTR " {
292			continue;
293		}
294
295		// Verify the basic checksum.
296		if verify_checksum(current_address, RSDP_CHECKSUM_LENGTH).is_err() {
297			debug!("Found an ACPI table at {current_address:#X}, but its RSDP checksum is invalid");
298			continue;
299		}
300
301		// Verify the extended checksum if this is an ACPI 2.0-compliant table.
302		if rsdp.revision >= 2 && verify_checksum(current_address, RSDP_XCHECKSUM_LENGTH).is_err() {
303			debug!(
304				"Found an ACPI table at {current_address:#X}, but its RSDP extended checksum is invalid"
305			);
306			continue;
307		}
308
309		// We were successful! Return a pointer to the RSDT (whose 64-bit address is called XSDT in this structure).
310		info!(
311			"Found an ACPI revision {} table at {:#X} with OEM ID \"{}\"",
312			rsdp.revision,
313			current_address,
314			rsdp.oem_id()
315		);
316		return Ok(rsdp);
317	}
318
319	// We found no valid ACPI RSDP.
320	Err(())
321}

323/// Detects ACPI support of the computer system.
324/// Returns a reference to the ACPI RSDP within the Ok() if successful or an empty Err() on failure.
325fn detect_acpi() -> Result<&'static AcpiRsdp, ()> {
326	if let Some(rsdp) = env::rsdp() {
327		trace!("RSDP detected successfully at {rsdp:#x?}");
328		let rsdp = unsafe {
329			ptr::with_exposed_provenance::<AcpiRsdp>(rsdp.get())
330				.as_ref()
331				.unwrap()
332		};
333		assert!(&rsdp.signature == b"RSD PTR ", "RSDP Address not valid!");
334		return Ok(rsdp);
335	}
336
337	// Get the address of the EBDA.
338	let frame = PhysFrame::<BasePageSize>::containing_address(EBDA_PTR_LOCATION.into());
339	paging::identity_map::<BasePageSize>(frame.start_address().into());
340	let ebda_ptr_location: &u16 =
341		unsafe { &*(VirtAddr::from(EBDA_PTR_LOCATION.as_u64()).as_ptr()) };
342	let ebda_address = PhysAddr::new(u64::from(*ebda_ptr_location) << 4);
343
344	// Check if the pointed address is valid. This check is also done in ACPICA.
345	if ebda_address > EBDA_MINIMUM_ADDRESS {
346		// Try to find an RSDP within the 1 KiB window of the EBDA.
347		if let Ok(rsdp) = detect_rsdp(ebda_address, ebda_address + EBDA_WINDOW_SIZE) {
348			return Ok(rsdp);
349		}
350	}
351
352	// If we didn't find anything above, check the other memory range for an RSDP.
353	if let Ok(rsdp) = detect_rsdp(RSDP_SEARCH_ADDRESS_LOW, RSDP_SEARCH_ADDRESS_HIGH) {
354		return Ok(rsdp);
355	}
356
357	// We didn't find any ACPI tables.
358	Err(())
359}

361fn search_s5_in_table(table: AcpiTable<'_>) {
362	// Get the AML code.
363	// As we do not implement an AML interpreter, we search through the bytecode.
364	let aml = unsafe {
365		slice::from_ptr_range(
366			ptr::with_exposed_provenance(table.table_start_address())
367				..ptr::with_exposed_provenance(table.table_end_address()),
368		)
369	};
370
371	// Find the "_S5_" object in the bytecode.
372	let s5 = [b'_', b'S', b'5', b'_', AML_PACKAGEOP];
373	let s5_position = aml.windows(s5.len()).position(|window| window == s5);
374	if let Some(i) = s5_position {
375		// We have found an "_S5_" object that looks valid.
376		// To be sure, verify that it begins with an AML_NAMEOP or an AML_NAMEOP and a backslash.
377		if i > 2 && (aml[i - 1] == AML_NAMEOP || (aml[i - 2] == AML_NAMEOP && aml[i - 1] == b'\\'))
378		{
379			// This is a valid "_S5_" object.
380			// It should be followed by this structure:
381			//    - single byte for PkgLength (index 5)
382			//    - single byte for NumElements (index 6)
383			let pkg_length = aml[i + 5];
384			let num_elements = aml[i + 6];
385
386			// Bits 6-7 of PkgLength are non-zero for larger packages, resulting in a different structure.
387			// This mustn't be the case for the "_S5_" object.
388			if pkg_length & 0b1100_0000 == 0 && num_elements > 0 {
389				// The next byte is an opcode describing the data.
390				// It is usually the byte prefix, indicating that the actual data is the single byte following the opcode.
391				// However, if the data is a zero or one byte, this may also be indicated by the opcode.
392				let op = aml[i + 7];
393				let slp_typa = match op {
394					AML_ZEROOP => 0,
395					AML_ONEOP => 1,
396					AML_BYTEPREFIX => aml[i + 8],
397					_ => return,
398				};
399
400				// All assumptions are correct, so slp_typa is supposed to contain valid information.
401				// Now we have all information we need for powering off through ACPI.
402				//
403				// Note that Power Off may also be controlled through PM1B_CNT_BLK / SLP_TYPB
404				// according to the ACPI Specification. However, this has not yet been observed on real computers
405				// and therefore not implemented.
406				SLP_TYPA.set(slp_typa).unwrap();
407			}
408		}
409	}
410}

412fn parse_fadt(fadt: AcpiTable<'_>) {
413	// Get us a reference to the actual fields of the FADT table.
414	// Note that not all fields may be accessible depending on the ACPI revision of the computer.
415	// Always check fadt.table_end_address() when accessing an optional field!
416	let fadt_table =
417		unsafe { &*ptr::with_exposed_provenance::<AcpiFadt>(fadt.table_start_address()) };
418
419	// Check if the FADT is large enough to hold an x_pm1a_cnt_blk field and if this field is non-zero.
420	// In that case, it shall be preferred over the I/O port specified in pm1a_cnt_blk.
421	// As all PM1 control registers are supposed to be in I/O space, we can simply check the address_space field
422	// of x_pm1a_cnt_blk to determine the validity of x_pm1a_cnt_blk.
423	let x_pm1a_cnt_blk_field_address = ptr::from_ref(&fadt_table.x_pm1a_cnt_blk).addr();
424	let pm1a_cnt_blk = if x_pm1a_cnt_blk_field_address < fadt.table_end_address()
425		&& fadt_table.x_pm1a_cnt_blk.address_space == GENERIC_ADDRESS_IO_SPACE
426	{
427		fadt_table.x_pm1a_cnt_blk.address as u16
428	} else {
429		fadt_table.pm1a_cnt_blk as u16
430	};
431	PM1A_CNT_BLK.set(Port::new(pm1a_cnt_blk)).unwrap();
432
433	// Map the "Differentiated System Description Table" (DSDT).
434	let x_dsdt_field_address = ptr::addr_of!(fadt_table.x_dsdt) as usize;
435	let dsdt_address = if x_dsdt_field_address < fadt.table_end_address() && fadt_table.x_dsdt > 0 {
436		PhysAddr::new(fadt_table.x_dsdt)
437	} else {
438		PhysAddr::new(fadt_table.dsdt.into())
439	};
440	let dsdt = AcpiTable::map(dsdt_address);
441
442	// Check it.
443	assert!(
444		dsdt.header.signature() == "DSDT",
445		"DSDT at {:p} has invalid signature \"{}\"",
446		dsdt_address,
447		dsdt.header.signature()
448	);
449	assert!(
450		verify_checksum(dsdt.header_start_address(), dsdt.header.length as usize).is_ok(),
451		"DSDT at {dsdt_address:p} has invalid checksum"
452	);
453
454	// Try to find the "_S5_" object for SLP_TYPA in the DSDT AML bytecode.
455	// It may also be in an SSDT though.
456	search_s5_in_table(dsdt);
457}

459fn parse_ssdt(ssdt: AcpiTable<'_>) {
460	// We don't need to parse the SSDT if we already have information about the "_S5_" object
461	// (e.g. from the DSDT or a previous SSDT).
462	if SLP_TYPA.get().is_some() {
463		return;
464	}
465
466	// Otherwise, just try to find "_S5_" information in the AML bytecode of this SSDT.
467	search_s5_in_table(ssdt);
468}

474pub fn poweroff() {
475	if let (Some(mut pm1a_cnt_blk), Some(&slp_typa)) = (PM1A_CNT_BLK.get().cloned(), SLP_TYPA.get())
476	{
477		let bits = (u16::from(slp_typa) << 10) | SLP_EN;
478		debug!("Powering Off through ACPI (port {pm1a_cnt_blk:?}, bitmask {bits:#X})");
479		unsafe {
480			pm1a_cnt_blk.write(bits);
481		}
482	} else {
483		warn!("ACPI Power Off is not available");
484	}
485}

487pub fn init() {
488	// Detect the RSDP and get a pointer to either the XSDT (64-bit) or RSDT (32-bit), whichever is available.
489	// Both are called RSDT in the following.
490	let rsdp = detect_acpi().expect("Hermit requires an ACPI-compliant system");
491	let rsdt_physical_address = if rsdp.revision >= 2 {
492		PhysAddr::new(rsdp.xsdt_physical_address)
493	} else {
494		PhysAddr::new(rsdp.rsdt_physical_address.into())
495	};
496
497	// Map the RSDT.
498	let rsdt = AcpiTable::map(rsdt_physical_address);
499
500	// The RSDT contains pointers to all available ACPI tables.
501	// Iterate through them.
502	let mut current_address = rsdt.table_start_address();
503	while current_address < rsdt.table_end_address() {
504		// Depending on the RSDP revision, either an XSDT or an RSDT has been chosen above.
505		// The XSDT contains 64-bit pointers whereas the RSDT has 32-bit pointers.
506		let table_physical_address = if rsdp.revision >= 2 {
507			let address = unsafe {
508				PhysAddr::new(ptr::with_exposed_provenance::<u64>(current_address).read_unaligned())
509			};
510			current_address += mem::size_of::<u64>();
511			address
512		} else {
513			let address = unsafe {
514				PhysAddr::new(
515					ptr::with_exposed_provenance::<u32>(current_address)
516						.read_unaligned()
517						.into(),
518				)
519			};
520			current_address += mem::size_of::<u32>();
521			address
522		};
523
524		let table = AcpiTable::map(table_physical_address);
525		debug!("Found ACPI table: {}", table.header.signature());
526
527		if table.header.signature() == "APIC" {
528			// The "Multiple APIC Description Table" (MADT) aka "APIC Table" (APIC)
529			// Check and save the entire APIC table for the get_apic_table() call.
530			assert!(
531				verify_checksum(table.header_start_address(), table.header.length as usize).is_ok(),
532				"MADT at {table_physical_address:p} has invalid checksum"
533			);
534			MADT.set(table).unwrap();
535		} else if table.header.signature() == "FACP" {
536			// The "Fixed ACPI Description Table" (FADT) aka "Fixed ACPI Control Pointer" (FACP)
537			// Check and parse this table for the poweroff() call.
538			assert!(
539				verify_checksum(table.header_start_address(), table.header.length as usize).is_ok(),
540				"FADT at {table_physical_address:p} has invalid checksum"
541			);
542			parse_fadt(table);
543		} else if table.header.signature() == "SSDT" {
544			assert!(
545				verify_checksum(table.header_start_address(), table.header.length as usize).is_ok(),
546				"SSDT at {table_physical_address:p} has invalid checksum"
547			);
