57pub struct RxQueues {
58	vqs: Vec<Box<dyn Virtq>>,
59	packet_size: u32,
60}
61
62impl RxQueues {
63	pub fn new(vqs: Vec<Box<dyn Virtq>>, dev_cfg: &NetDevCfg) -> Self {
64		// See Virtio specification v1.1 - 5.1.6.3.1
65		//
66		let packet_size = if dev_cfg.features.contains(virtio::net::F::MRG_RXBUF) {
67			1514
68		} else {
69			dev_cfg.raw.as_ptr().mtu().read().to_ne().into()
70		};
71
72		Self { vqs, packet_size }
73	}
74
75	/// Takes care of handling packets correctly which need some processing after being received.
76	/// This currently include nothing. But in the future it might include among others:
77	/// * Calculating missing checksums
78	/// * Merging receive buffers, by simply checking the poll_queue (if VIRTIO_NET_F_MRG_BUF)
79	fn post_processing(_buffer_tkn: &mut UsedBufferToken) -> Result<(), VirtioNetError> {
80		Ok(())
81	}
82
83	/// Adds a given queue to the underlying vector and populates the queue with RecvBuffers.
84	///
85	/// Queues are all populated according to Virtio specification v1.1. - 5.1.6.3.1
86	fn add(&mut self, mut vq: Box<dyn Virtq>) {
87		const BUFF_PER_PACKET: u16 = 2;
88		let num_packets: u16 = u16::from(vq.size()) / BUFF_PER_PACKET;
89		fill_queue(vq.as_mut(), num_packets, self.packet_size);
90		self.vqs.push(vq);
91	}
92
93	fn get_next(&mut self) -> Option<UsedBufferToken> {
94		self.vqs[0].try_recv().ok()
95	}
96
97	fn enable_notifs(&mut self) {
98		for vq in &mut self.vqs {
99			vq.enable_notifs();
100		}
101	}
102
103	fn disable_notifs(&mut self) {
104		for vq in &mut self.vqs {
105			vq.disable_notifs();
106		}
107	}
108
109	fn has_packet(&self) -> bool {
110		self.vqs.iter().any(|vq| vq.has_used_buffers())
111	}
112}
113
114fn fill_queue(vq: &mut dyn Virtq, num_packets: u16, packet_size: u32) {
115	for _ in 0..num_packets {
116		let buff_tkn = match AvailBufferToken::new(
117			vec![],
118			vec![
119				BufferElem::Sized(Box::<Hdr, _>::new_uninit_in(DeviceAlloc)),
120				BufferElem::Vector(Vec::with_capacity_in(
121					packet_size.try_into().unwrap(),
122					DeviceAlloc,
123				)),
124			],
125		) {
126			Ok(tkn) => tkn,
127			Err(_vq_err) => {
128				error!("Setup of network queue failed, which should not happen!");
129				panic!("setup of network queue failed!");
130			}
131		};
132
133		// BufferTokens are directly provided to the queue
134		// TransferTokens are directly dispatched
135		// Transfers will be awaited at the queue
136		if let Err(err) = vq.dispatch(buff_tkn, false, BufferType::Direct) {
137			error!("{err:#?}");
138			break;
139		}
140	}
141}

143/// Structure which handles transmission of packets and delegation
144/// to the respective queue structures.
145pub struct TxQueues {
146	vqs: Vec<Box<dyn Virtq>>,
147	/// Indicates, whether the Driver/Device are using multiple
148	/// queues for communication.
149	packet_length: u32,
150}
151
152impl TxQueues {
153	pub fn new(vqs: Vec<Box<dyn Virtq>>, dev_cfg: &NetDevCfg) -> Self {
154		let packet_length = if dev_cfg.features.contains(virtio::net::F::GUEST_TSO4)
155			| dev_cfg.features.contains(virtio::net::F::GUEST_TSO6)
156			| dev_cfg.features.contains(virtio::net::F::GUEST_UFO)
157		{
158			0x0001_000e
159		} else {
160			dev_cfg.raw.as_ptr().mtu().read().to_ne().into()
161		};
162
163		Self { vqs, packet_length }
164	}
165	#[allow(dead_code)]
166	fn enable_notifs(&mut self) {
167		for vq in &mut self.vqs {
168			vq.enable_notifs();
169		}
170	}
171
172	#[allow(dead_code)]
173	fn disable_notifs(&mut self) {
174		for vq in &mut self.vqs {
175			vq.disable_notifs();
176		}
177	}
178
179	fn poll(&mut self) {
180		for vq in &mut self.vqs {
181			// We don't do anything with the buffers but we need to receive them for the
182			// ring slots to be emptied and the memory from the previous transfers to be freed.
183			while vq.try_recv().is_ok() {}
184		}
185	}
186
187	fn add(&mut self, vq: Box<dyn Virtq>) {
188		// Currently we are doing nothing with the additional queues. They are inactive and might be used in the
189		// future
190		self.vqs.push(vq);
191	}
192}

241	/// Provides smoltcp a slice to copy the IP packet and transfer the packet
242	/// to the send queue.
243	fn send_packet<R, F>(&mut self, len: usize, f: F) -> R
244	where
245		F: FnOnce(&mut [u8]) -> R,
246	{
247		// We need to poll to get the queue to remove elements from the table and make space for
248		// what we are about to add
249		self.send_vqs.poll();
250
251		assert!(len < usize::try_from(self.send_vqs.packet_length).unwrap());
252		let mut packet = Vec::with_capacity_in(len, DeviceAlloc);
253		let result = unsafe {
254			let result = f(packet.spare_capacity_mut().assume_init_mut());
255			packet.set_len(len);
256			result
257		};
258
259		let mut header = Box::new_in(<Hdr as Default>::default(), DeviceAlloc);
260		// If a checksum isn't necessary, we have inform the host within the header
261		// see Virtio specification 5.1.6.2
262		if !self.checksums.tcp.tx() || !self.checksums.udp.tx() {
263			header.flags = HdrF::NEEDS_CSUM;
264			let ethernet_frame: smoltcp::wire::EthernetFrame<&[u8]> =
265				EthernetFrame::new_unchecked(&packet);
266			let packet_header_len: u16;
267			let protocol;
268			match ethernet_frame.ethertype() {
269				smoltcp::wire::EthernetProtocol::Ipv4 => {
270					let packet = Ipv4Packet::new_unchecked(ethernet_frame.payload());
271					packet_header_len = packet.header_len().into();
272					protocol = Some(packet.next_header());
273				}
274				smoltcp::wire::EthernetProtocol::Ipv6 => {
275					let packet = Ipv6Packet::new_unchecked(ethernet_frame.payload());
276					packet_header_len = packet.header_len().try_into().unwrap();
277					protocol = Some(packet.next_header());
278				}
279				_ => {
280					packet_header_len = 0;
281					protocol = None;
282				}
283			}
284			header.csum_start =
285				(u16::try_from(ETHERNET_HEADER_LEN).unwrap() + packet_header_len).into();
286			header.csum_offset = match protocol {
287				Some(smoltcp::wire::IpProtocol::Tcp) => 16,
288				Some(smoltcp::wire::IpProtocol::Udp) => 6,
289				_ => 0,
290			}
291			.into();
292		}
293
294		let buff_tkn = AvailBufferToken::new(
295			vec![BufferElem::Sized(header), BufferElem::Vector(packet)],
296			vec![],
297		)
298		.unwrap();
299
300		self.send_vqs.vqs[0]
301			.dispatch(buff_tkn, false, BufferType::Direct)
302			.unwrap();
303
304		result
305	}
306
307	fn receive_packet(&mut self) -> Option<(RxToken, TxToken)> {
308		let mut receive_single_packet = || {
309			let mut buffer_tkn = self.recv_vqs.get_next()?;
310			RxQueues::post_processing(&mut buffer_tkn)
311				.inspect_err(|vnet_err| warn!("Post processing failed. Err: {vnet_err:?}"))
312				.ok()?;
313			let header = buffer_tkn.used_recv_buff.pop_front_downcast::<Hdr>()?;
314			let packet = buffer_tkn.used_recv_buff.pop_front_vec()?;
315			Some((header, packet))
316		};
317
318		let (first_header, first_packet) = receive_single_packet()?;
319
320		// According to VIRTIO spec v1.2 sec. 5.1.6.3.2, "num_buffers will always be 1 if VIRTIO_NET_F_MRG_RXBUF is not negotiated."
321		// Unfortunately, NVIDIA MLX5 does not comply with this requirement and we have to manually set the value to the correct one.
322		let num_buffers = if self.dev_cfg.features.contains(virtio::net::F::MRG_RXBUF) {
323			first_header.num_buffers.to_ne()
324		} else {
325			1
326		};
327
328		let mut combined_packets = first_packet;
329
330		for _ in 1..num_buffers {
331			let (_header, packet) = receive_single_packet()?;
332			combined_packets.extend_from_slice(&packet);
333		}
334
335		fill_queue(
336			self.recv_vqs.vqs[0].as_mut(),
337			num_buffers,
338			self.recv_vqs.packet_size,
339		);
340
341		Some((RxToken::new(combined_packets), TxToken::new()))
342	}
343
344	fn set_polling_mode(&mut self, value: bool) {
345		if value {
346			self.disable_interrupts();
347		} else {
348			self.enable_interrupts();
349		}
350	}

352	fn handle_interrupt(&mut self) {
353		let status = self.isr_stat.is_queue_interrupt();
354
355		#[cfg(not(feature = "pci"))]
356		if status.contains(virtio::mmio::InterruptStatus::CONFIGURATION_CHANGE_NOTIFICATION) {
357			info!("Configuration changes are not possible! Aborting");
358			todo!("Implement possibility to change config on the fly...")
359		}
360
361		#[cfg(feature = "pci")]
362		if status.contains(virtio::pci::IsrStatus::DEVICE_CONFIGURATION_INTERRUPT) {
363			info!("Configuration changes are not possible! Aborting");
364			todo!("Implement possibility to change config on the fly...")
365		}
366
367		self.isr_stat.acknowledge();
368	}

465	/// Initializes the device in adherence to specification. Returns Some(VirtioNetError)
466	/// upon failure and None in case everything worked as expected.
467	///
468	/// See Virtio specification v1.1. - 3.1.1.
469	///                      and v1.1. - 5.1.5
470	pub fn init_dev(&mut self) -> Result<(), VirtioNetError> {
471		// Reset
472		self.com_cfg.reset_dev();
473
474		// Indicate device, that OS noticed it
475		self.com_cfg.ack_dev();
476
477		// Indicate device, that driver is able to handle it
478		self.com_cfg.set_drv();
479
480		let minimal_features = virtio::net::F::VERSION_1 | virtio::net::F::MAC;
481
482		// If wanted, push new features into feats here:
483		let mut features = minimal_features
484			// Indirect descriptors can be used
485			| virtio::net::F::INDIRECT_DESC
486			// Packed Vq can be used
487			| virtio::net::F::RING_PACKED
488			| virtio::net::F::NOTIFICATION_DATA
489			// Host should avoid the creation of checksums
490			| virtio::net::F::CSUM
491			// Guest avoids the creation of checksums
492			| virtio::net::F::GUEST_CSUM
493			// MTU setting can be used
494			| virtio::net::F::MTU
495			// Driver can merge receive buffers
496			| virtio::net::F::MRG_RXBUF
497			// the link status can be announced
498			| virtio::net::F::STATUS
499			// control queue support
500			| virtio::net::F::CTRL_VQ
501			// Multiqueue support
502			| virtio::net::F::MQ;
503
504		// Currently the driver does NOT support the features below.
505		// In order to provide functionality for these, the driver
506		// needs to take care of calculating checksum in
507		// RxQueues.post_processing()
508		// | virtio::net::F::GUEST_TSO4
509		// | virtio::net::F::GUEST_TSO6
510
511		// Negotiate features with device. Automatically reduces selected feats in order to meet device capabilities.
512		// Aborts in case incompatible features are selected by the driver or the device does not support min_feat_set.
513		match self.negotiate_features(features) {
514			Ok(()) => info!(
515				"Driver found a subset of features for virtio device {:x}. Features are: {features:?}",
516				self.dev_cfg.dev_id
517			),
518			Err(vnet_err) => {
519				match vnet_err {
520					VirtioNetError::FeatureRequirementsNotMet(features) => {
521						error!(
522							"Network drivers feature set {features:?} does not satisfy rules in section 5.1.3.1 of specification v1.1. Aborting!"
523						);
524						return Err(vnet_err);
525					}
526					VirtioNetError::IncompatibleFeatureSets(drv_feats, dev_feats) => {
527						// Create a new matching feature set for device and driver if the minimal set is met!
528						if !dev_feats.contains(minimal_features) {
529							error!(
530								"Device features set, does not satisfy minimal features needed. Aborting!"
531							);
532							return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
533						}
534
535						let common_features = drv_feats & dev_feats;
536						if common_features.is_empty() {
537							error!(
538								"Feature negotiation failed with minimal feature set. Aborting!"
539							);
540							return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
541						}
542						features = common_features;
543
544						match self.negotiate_features(features) {
545							Ok(()) => info!(
546								"Driver found a subset of features for virtio device {:x}. Features are: {features:?}",
547								self.dev_cfg.dev_id
548							),
549							Err(vnet_err) => match vnet_err {
550								VirtioNetError::FeatureRequirementsNotMet(features) => {
551									error!(
552										"Network device offers a feature set {features:?} when used completely does not satisfy rules in section 5.1.3.1 of specification v1.1. Aborting!"
553									);
554									return Err(vnet_err);
555								}
556								_ => {
557									error!(
558										"Feature Set after reduction still not usable. Set: {features:?}. Aborting!"
559									);
560									return Err(vnet_err);
561								}
562							},
563						}
564					}
565					VirtioNetError::FailFeatureNeg(_) => {
566						error!(
567							"Wanted set of features is NOT supported by device. Set: {features:?}"
568						);
569						return Err(vnet_err);
570					}
571					#[cfg(feature = "pci")]
572					VirtioNetError::NoDevCfg(_) => {
573						error!("No device config found.");
574						return Err(vnet_err);
575					}
576				}
577			}
578		}
579
580		// Indicates the device, that the current feature set is final for the driver
581		// and will not be changed.
582		self.com_cfg.features_ok();
583
584		// Checks if the device has accepted final set. This finishes feature negotiation.
585		if self.com_cfg.check_features() {
586			info!(
587				"Features have been negotiated between virtio network device {:x} and driver.",
588				self.dev_cfg.dev_id
589			);
590			// Set feature set in device config fur future use.
591			self.dev_cfg.features = features;
592		} else {
593			return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
594		}
595
596		self.dev_spec_init()?;
597		info!(
598			"Device specific initialization for Virtio network device {:x} finished",
599			self.dev_cfg.dev_id
600		);
601
602		// At this point the device is "live"
603		self.com_cfg.drv_ok();
604
605		if self.dev_cfg.features.contains(virtio::net::F::CSUM)
606			&& self.dev_cfg.features.contains(virtio::net::F::GUEST_CSUM)
607		{
608			self.checksums.udp = Checksum::None;
609			self.checksums.tcp = Checksum::None;
610		} else if self.dev_cfg.features.contains(virtio::net::F::CSUM) {
611			self.checksums.udp = Checksum::Rx;
612			self.checksums.tcp = Checksum::Rx;
613		} else if self.dev_cfg.features.contains(virtio::net::F::GUEST_CSUM) {
614			self.checksums.udp = Checksum::Tx;
615			self.checksums.tcp = Checksum::Tx;
616		}
617		debug!("{:?}", self.checksums);
618
619		if self.dev_cfg.features.contains(virtio::net::F::MTU) {
620			self.mtu = self.dev_cfg.raw.as_ptr().mtu().read().to_ne();
621		}
622
623		Ok(())
624	}
625
626	/// Negotiates a subset of features, understood and wanted by both the OS
627	/// and the device.
628	fn negotiate_features(
629		&mut self,
630		driver_features: virtio::net::F,
631	) -> Result<(), VirtioNetError> {
632		let device_features = virtio::net::F::from(self.com_cfg.dev_features());
633
634		if device_features.requirements_satisfied() {
635			info!("Feature set wanted by network driver are in conformance with specification.");
636		} else {
637			return Err(VirtioNetError::FeatureRequirementsNotMet(device_features));
638		}
639
640		if device_features.contains(driver_features) {
641			// If device supports subset of features write feature set to common config
642			self.com_cfg.set_drv_features(driver_features.into());
643			Ok(())
644		} else {
645			Err(VirtioNetError::IncompatibleFeatureSets(
646				driver_features,
647				device_features,
648			))
649		}
650	}
651
652	/// Device Specific initialization according to Virtio specifictation v1.1. - 5.1.5
653	fn dev_spec_init(&mut self) -> Result<(), VirtioNetError> {
654		self.virtqueue_init()?;
655		info!("Network driver successfully initialized virtqueues.");
656
657		// Add a control if feature is negotiated
658		if self.dev_cfg.features.contains(virtio::net::F::CTRL_VQ) {
659			if self.dev_cfg.features.contains(virtio::net::F::RING_PACKED) {
660				self.ctrl_vq = CtrlQueue(Some(Box::new(
661					PackedVq::new(
662						&mut self.com_cfg,
663						&self.notif_cfg,
664						VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
665						VqIndex::from(self.num_vqs),
666						self.dev_cfg.features.into(),
667					)
668					.unwrap(),
669				)));
670			} else {
671				self.ctrl_vq = CtrlQueue(Some(Box::new(
672					SplitVq::new(
673						&mut self.com_cfg,
674						&self.notif_cfg,
675						VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
676						VqIndex::from(self.num_vqs),
677						self.dev_cfg.features.into(),
678					)
679					.unwrap(),
680				)));
681			}
682
683			self.ctrl_vq.0.as_mut().unwrap().enable_notifs();
684		}
685
686		Ok(())
687	}
688
689	/// Initialize virtqueues via the queue interface and populates receiving queues
690	fn virtqueue_init(&mut self) -> Result<(), VirtioNetError> {
691		// We are assuming here, that the device single source of truth is the
692		// device specific configuration. Hence we do NOT check if
693		//
694		// max_virtqueue_pairs + 1 < num_queues
695		//
696		// - the plus 1 is due to the possibility of an existing control queue
697		// - the num_queues is found in the ComCfg struct of the device and defines the maximal number
698		// of supported queues.
699		if self.dev_cfg.features.contains(virtio::net::F::MQ) {
700			if self
701				.dev_cfg
702				.raw
703				.as_ptr()
704				.max_virtqueue_pairs()
705				.read()
706				.to_ne() * 2 >= MAX_NUM_VQ
707			{
708				self.num_vqs = MAX_NUM_VQ;
709			} else {
710				self.num_vqs = self
711					.dev_cfg
712					.raw
713					.as_ptr()
714					.max_virtqueue_pairs()
715					.read()
716					.to_ne() * 2;
717			}
718		} else {
719			// Minimal number of virtqueues defined in the standard v1.1. - 5.1.5 Step 1
720			self.num_vqs = 2;
721		}
722
723		// The loop is running from 0 to num_vqs and the indexes are provided to the VqIndex::from function in this way
724		// in order to allow the indexes of the queues to be in a form of:
725		//
726		// index i for receive queue
727		// index i+1 for send queue
728		//
729		// as it is wanted by the network network device.
730		// see Virtio specification v1.1. - 5.1.2
731		// Assure that we have always an even number of queues (i.e. pairs of queues).
732		assert_eq!(self.num_vqs % 2, 0);
733
734		for i in 0..(self.num_vqs / 2) {
735			if self.dev_cfg.features.contains(virtio::net::F::RING_PACKED) {
736				let mut vq = PackedVq::new(
737					&mut self.com_cfg,
738					&self.notif_cfg,
739					VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
740					VqIndex::from(2 * i),
741					self.dev_cfg.features.into(),
742				)
743				.unwrap();
744				// Interrupt for receiving packets is wanted
745				vq.enable_notifs();
746
747				self.recv_vqs.add(Box::from(vq));
748
749				let mut vq = PackedVq::new(
750					&mut self.com_cfg,
751					&self.notif_cfg,
752					VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
753					VqIndex::from(2 * i + 1),
754					self.dev_cfg.features.into(),
755				)
756				.unwrap();
757				// Interrupt for communicating that a sended packet left, is not needed
758				vq.disable_notifs();
759
760				self.send_vqs.add(Box::from(vq));
761			} else {
762				let mut vq = SplitVq::new(
763					&mut self.com_cfg,
764					&self.notif_cfg,
765					VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
766					VqIndex::from(2 * i),
767					self.dev_cfg.features.into(),
768				)
769				.unwrap();
770				// Interrupt for receiving packets is wanted
771				vq.enable_notifs();
772
773				self.recv_vqs.add(Box::from(vq));
774
775				let mut vq = SplitVq::new(
776					&mut self.com_cfg,
777					&self.notif_cfg,
778					VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
779					VqIndex::from(2 * i + 1),
780					self.dev_cfg.features.into(),
781				)
782				.unwrap();
783				// Interrupt for communicating that a sended packet left, is not needed
784				vq.disable_notifs();
785
786				self.send_vqs.add(Box::from(vq));
787			}
788		}
789
790		Ok(())
791	}
792}
